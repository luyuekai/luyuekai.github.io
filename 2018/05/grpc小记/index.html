GRPC小记

五月 27, 2018 发布在 Code

有一种落差是你配不上自己的野心，也辜负了所受的苦难。

在看分布式书籍及源码解析时，总是可以看到好多分布式框架，如： Hadoop,Spark等。都自身实现了一套自己的RPC机制。 RPC这东西，在看技术博客时，经常能看到，自己也去了解了很多。

RPC又称远程过程调用（Remote Procedure Call)。 个人理解， 就是一个计算机要去调用另一个计算机上的服务。粗一看，感觉和REST差不多嘛。 其实 RPC和REST之间确实有相似相同之处。RPC更强调快速，高并发吧。 通常会选择采用什么传输协议（TCP,HTTP,还是其他）。 REST一般都是http来传输的。

参考个链接 RPC VS REST

常见的，也经常被使用的RPC框架： Thrift,gRPC. 其他的待补充吧。
今天在intellij中跑通了 gRPC的一个例子。 把过程记录一下，万一以后会遇到呢~ 嘻嘻

  • 三、 在main文件夹下创建proto文件夹,专门用来放.proto的文件。
  • 四、 在proto的文件夹中创建grpc.proto(new 时选择file,自己来添加后缀名)
    grpc.proto的内容:
syntax = "proto3";

option java_multiple_files = true;
option java_package = "io.grpc.examples.helloworld";
option java_outer_classname = "HelloWorldProto";
option objc_class_prefix = "HLW";

package helloworld;

// The greeting service definition.
service Greeter {
    // Sends a greeting
    rpc SayHello (HelloRequest) returns (HelloReply) {}
}

// The request message containing the user's name.
message HelloRequest {
    string name = 1;
}

// The response message containing the greetings
message HelloReply {
    string message = 1;
}

  • 五、 点击右侧 maven compile（第一次运行直接compile就行，以后可能需要先clean）

  • 六、 在 java目录下，创建HelloWorldClient.java和HelloWorldServer.java文件，文件内容

HelloWorldClient.java的文件内容：

package com;

import io.grpc.ManagedChannel;
import io.grpc.ManagedChannelBuilder;
import io.grpc.examples.helloworld.GreeterGrpc;
import io.grpc.examples.helloworld.HelloReply;
import io.grpc.examples.helloworld.HelloRequest;
import java.util.concurrent.TimeUnit;


public class HelloWorldClient {

    private final ManagedChannel channel;
    private final GreeterGrpc.GreeterBlockingStub blockingStub;


    public HelloWorldClient(String host,int port){
        channel = ManagedChannelBuilder.forAddress(host,port)
                .usePlaintext()
                .build();

        blockingStub = GreeterGrpc.newBlockingStub(channel);
    }


    public void shutdown() throws InterruptedException {
        channel.shutdown().awaitTermination(5, TimeUnit.SECONDS);
    }

    public  void greet(String name){
        HelloRequest request = HelloRequest.newBuilder().setName(name).build();
        HelloReply   response = blockingStub.sayHello(request);
        System.out.println(response.getMessage());

    }

    public static void main(String[] args) throws InterruptedException {
        HelloWorldClient client = new HelloWorldClient("127.0.0.1",50051);
        for(int i=0;i<5;i++){
            client.greet("world:"+i);
        }
        client.shutdown();

    }
}

HelloWorldServer.java的文件内容：

package com;

import io.grpc.Server;
import io.grpc.ServerBuilder;
import io.grpc.examples.helloworld.GreeterGrpc;
import io.grpc.examples.helloworld.HelloReply;
import io.grpc.examples.helloworld.HelloRequest;
import io.grpc.stub.StreamObserver;

import java.io.IOException;

public class HelloWorldServer {


    private int port = 50051;
    private Server server;

    private void start() throws IOException {
        server = ServerBuilder.forPort(port)
                .addService(new GreeterImpl())
                .build()
                .start();

        System.out.println("service start...");

        Runtime.getRuntime().addShutdownHook(new Thread() {

            @Override
            public void run() {

                System.err.println("*** shutting down gRPC server since JVM is shutting down");
                HelloWorldServer.this.stop();
                System.err.println("*** server shut down");
            }
        });
    }

    private void stop() {
        if (server != null) {
            server.shutdown();
        }
    }

    // block 一直到退出程序
    private void blockUntilShutdown() throws InterruptedException {
        if (server != null) {
            server.awaitTermination();
        }
    }


    public static void main(String[] args) throws IOException, InterruptedException {

        final HelloWorldServer server = new HelloWorldServer();
        server.start();
        server.blockUntilShutdown();
    }


    // 实现 定义一个实现服务接口的类
    private class GreeterImpl extends GreeterGrpc.GreeterImplBase {


        public void sayHello(HelloRequest req, StreamObserver<HelloReply> responseObserver) {
            System.out.println("service:"+req.getName());
            HelloReply reply = HelloReply.newBuilder().setMessage(("Hello: " + req.getName())).build();
            responseObserver.onNext(reply);
            responseObserver.onCompleted();
        }
    }
}
